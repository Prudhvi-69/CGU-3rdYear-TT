import java.util.Scanner;
import java.util.Stack;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        Stack<Integer> stack = new Stack<>();
        Stack<Integer> reverse = new Stack<>();

        while (sc.hasNextInt()) {
            stack.push(sc.nextInt());
        }

        while (!stack.isEmpty()) {
            reverse.push(stack.pop());
        }

        while (!reverse.isEmpty()) {
            System.out.print(reverse.peek() + " ");
            reverse.pop();
        }
    }
}












import java.util.Scanner;
import java.util.Stack;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        Stack<Integer> stack = new Stack<>();
        Stack<Integer> temp = new Stack<>();

        while (true) {
            int x = sc.nextInt();
            if (x == -1)
                break;
            stack.push(x);
        }

        int value = sc.nextInt();

        while (!stack.isEmpty()) {
            temp.push(stack.pop());
        }

        stack.push(value);

        while (!temp.isEmpty()) {
            stack.push(temp.pop());
        }

        System.out.println("Before Pushing :");

        for (int i = stack.size() - 1; i >= 1; i--) {
            System.out.print(stack.get(i) + " ");
        }

        System.out.println();

        System.out.println("After Pushing :");

        for (int i = stack.size() - 1; i >= 0; i--) {
            System.out.print(stack.get(i) + " ");
        }
    }
}











import java.util.Scanner;
import java.util.Stack;

public class Main {

    static void reverse(Stack<Integer> stack) {
        if (stack.isEmpty())
            return;

        int val = stack.pop();
        reverse(stack);
        System.out.print(val + " ");
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        Stack<Integer> stack = new Stack<>();

        while (sc.hasNextInt()) {
            stack.push(sc.nextInt());
        }

        reverse(stack);
    }
}

import java.util.Scanner;
import java.util.Stack;

public class Main {

    static void pushBottom(Stack<Integer> stack, int value) {
        if (stack.isEmpty()) {
            stack.push(value);
            return;
        }

        int temp = stack.pop();
        pushBottom(stack, value);
        stack.push(temp);
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        Stack<Integer> stack = new Stack<>();

        while (true) {
            int x = sc.nextInt();
            if (x == -1)
                break;
            stack.push(x);
        }

        int value = sc.nextInt();

        pushBottom(stack, value);

        for (int i = stack.size() - 1; i >= 0; i--) {
            System.out.print(stack.get(i) + " ");
        }
    }
}
