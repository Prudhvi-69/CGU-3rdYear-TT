// Q1 : Population Migration
// Input : N (population), years -> Output : final population

// Python

n = int(input())
years = int(input())
i = 1
while i <= years:
    n = n - n * 20 // 100
    n += 15
    if i % 4 == 0:
        n -= 10
    i += 1
print(n)


// Java

import java.util.Scanner;
public class Q1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int years = sc.nextInt();
        for (int i = 1; i <= years; i++) {
            n = n - n * 20 / 100;
            n += 15;
            if (i % 4 == 0) n -= 10;
        }
        System.out.println(n);
    }
}

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int months = sc.nextInt();
        int a = 100, b = 150;
        if (months == 1) { System.out.println(a); return; }
        if (months == 2) { System.out.println(b); return; }

            if (i % 3 == 0) c -= 20;
            a = b; b = c;

        System.out.println(b);
    }

// Q5 : Monster Population
// Input : n -> Output : monsters at generation n

// Python

n = int(input())
vals = [0, 1, 3]
i = 3
while i <= n:
    v = vals[i-1] + vals[i-2]
    prime = 1
    if i < 2:
        prime = 0
    j = 2
    while j * j <= i:
        if i % j == 0:
            prime = 0
        j += 1
    if prime == 1:
        v = v - v * 10 // 100
    vals.append(v)
    i += 1
print(vals[n])


// Java

import java.util.Scanner;
public class Q5 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[] vals = new int[n + 1];
        vals[1] = 1;
        if (n >= 2) vals[2] = 3;
        for (int i = 3; i <= n; i++) {
            int v = vals[i-1] + vals[i-2];
            boolean prime = true;
            if (i < 2) prime = false;
            for (int j = 2; j * j <= i; j++)
                if (i % j == 0) { prime = false; break; }
            if (prime) v = v - v * 10 / 100;
            vals[i] = v;
        }
        System.out.println(vals[n]);
    }
}




// Q6 : Galactic Fuel
// Input:1->1, Input:4->5, Input:8->56

// Python

n = int(input())
vals = [1, 1, 2, 3]
if n <= 3:
    print(vals[n])
else:
    i = 4
    while i <= n:
        v = 2 * vals[i-1] - vals[i-4]
        if v % 9 == 0:
            v -= 1
        vals.append(v)
        i += 1
    print(vals[n])


// Java

import java.util.Scanner;
public class Q6 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[] vals = new int[n + 1];
        vals[0] = 1; vals[1] = 1; vals[2] = 2; vals[3] = 3;
        for (int i = 4; i <= n; i++) {
            int v = 2 * vals[i-1] - vals[i-4];
            if (v % 9 == 0) v -= 1;
            vals[i] = v;
        }
        System.out.println(vals[n]);
    }
}
