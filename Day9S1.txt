// Q1 : Count words in string
// Input : "Book Inglis LopTap EhSea Pyan Pookies ohkhe" -> Output : 7

// Python

s = input()
count = 0
i = 0
while i < len(s):
    if s[i] != ' ' and (i == 0 or s[i-1] == ' '):
        count += 1
    i += 1
print(count)


// Java

import java.util.Scanner;
public class Q1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String s = sc.nextLine();
        int count = 0;
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) != ' ' && (i == 0 || s.charAt(i-1) == ' '))
                count++;
        }
        System.out.println(count);
    }
}




// Q2 : Sort words by length (stable - keep original order for same length)
// Input : "Book Inglis LopTap EhSea Pyan Pookies ohkhe"
// Output : "Book Pyan EhSea ohkhe Inglis LopTap Pookies"

// Python

s = input()
words = [""] * 100
n = 0
word = ""
i = 0
while i <= len(s):
    if i == len(s) or s[i] == ' ':
        if len(word) > 0:
            words[n] = word
            n += 1
            word = ""
    else:
        word += s[i]
    i += 1

# Bubble sort by length (stable)
i = 0
while i < n - 1:
    j = 0
    while j < n - i - 1:
        if len(words[j]) > len(words[j+1]):
            temp = words[j]
            words[j] = words[j+1]
            words[j+1] = temp
        j += 1
    i += 1

result = ""
i = 0
while i < n:
    result += words[i]
    if i != n - 1:
        result += " "
    i += 1
print(result)


// Java

import java.util.Scanner;
public class Q2 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String s = sc.nextLine();
        String[] words = new String[100];
        int n = 0;
        String word = "";
        for (int i = 0; i <= s.length(); i++) {
            if (i == s.length() || s.charAt(i) == ' ') {
                if (word.length() > 0) { words[n++] = word; word = ""; }
            } else {
                word += s.charAt(i);
            }
        }
        // Bubble sort by length (stable)
        for (int i = 0; i < n - 1; i++)
            for (int j = 0; j < n - i - 1; j++)
                if (words[j].length() > words[j+1].length()) {
                    String temp = words[j]; words[j] = words[j+1]; words[j+1] = temp;
                }
        String result = "";
        for (int i = 0; i < n; i++) {
            if (i > 0) result += " ";
            result += words[i];
        }
        System.out.println(result);
    }
}
