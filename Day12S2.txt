At the end of the Heist, you need to identify which Light Beams are active.

import java.util.Scanner;
import java.util.Stack;

public class Main {
    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);
        String[] arr = sc.nextLine().split(" ");

        Stack<String> stack = new Stack<>();
        Stack<String> temp = new Stack<>();
        Stack<String> leaked = new Stack<>();

        for (String beam : arr) {

            boolean skip = false;
            for (String x : leaked) {
                if (x.equals(beam)) {
                    skip = true;
                    break;
                }
            }

            if (skip)
                continue;

            if (stack.isEmpty()) {
                stack.push(beam);
                continue;
            }

            if (stack.peek().equals(beam)) {
                stack.pop();
                continue;
            }

            boolean found = false;

            while (!stack.isEmpty()) {

                if (stack.peek().equals(beam)) {
                    found = true;
                    stack.pop();
                    break;
                }

                temp.push(stack.pop());
            }

            if (found) {

                while (!temp.isEmpty()) {

                    String x = temp.pop();

                    boolean exist = false;
                    for (String y : leaked) {
                        if (y.equals(x)) {
                            exist = true;
                            break;
                        }
                    }

                    if (!exist)
                        leaked.push(x);
                }

            } else {

                while (!temp.isEmpty())
                    stack.push(temp.pop());

                stack.push(beam);
            }
        }

        Stack<String> result = new Stack<>();

        for (String x : stack)
            result.push(x);

        for (String x : leaked) {

            boolean exist = false;

            for (String y : result) {
                if (y.equals(x)) {
                    exist = true;
                    break;
                }
            }

            if (!exist)
                result.push(x);
        }

        for (String x : result)
            System.out.print(x + " ");
    }
}
