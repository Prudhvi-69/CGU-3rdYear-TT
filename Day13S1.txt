import java.util.Scanner;

class StackArray {
    int arr[];
    int top;
    int size;

    StackArray(int size) {
        this.size = size;
        arr = new int[size];
        top = -1;
    }

    void push(int value) {
        if (isFull()) {
            System.out.println("Stack Overflow");
            return;
        }
        arr[++top] = value;
    }

    void pop() {
        if (isEmpty()) {
            System.out.println("Stack Underflow");
            return;
        }
        System.out.println("Popped : " + arr[top--]);
    }

    void peek() {
        if (isEmpty()) {
            System.out.println("Stack Empty");
            return;
        }
        System.out.println(arr[top]);
    }

    void access(int index) {
        if (index > top) {
            System.out.println("Out of Range");
        } else if (index >= 0 && index < top) {
            System.out.println("RESTRICTED ACCESS");
        } else if (index == top) {
            System.out.println(arr[index]);
        } else {
            System.out.println("Invalid Index");
        }
    }

    void display() {
        if (isEmpty()) {
            System.out.println("Stack Empty");
            return;
        }

        while (!isEmpty()) {
            System.out.print(arr[top] + " ");
            top--;
        }
        System.out.println();
    }

    boolean isEmpty() {
        return top == -1;
    }

    boolean isFull() {
        return top == size - 1;
    }
}

        self.size = size
        self.top = -1

    def isFull(self):

    def isEmpty(self):
        return self.top == -1

    def push(self, value):
        if self.isFull():
            print("Stack Overflow")
            return
        self.top += 1
        self.arr[self.top] = value

    def pop(self):
        if self.isEmpty():
            print("Stack Underflow")
            return
        print("Popped :", self.arr[self.top])
        self.top -= 1

    def peek(self):
        if self.isEmpty():
            print("Stack Empty")
            return
        print(self.arr[self.top])

    def access(self, index):
        if index > self.top:
            print("Out of Range")
        elif index >= 0 and index < self.top:
            print("RESTRICTED ACCESS")
        elif index == self.top:
            print(self.arr[index])
        else:
            print("Invalid Index")

    def display(self):
        if self.isEmpty():
            print("Stack Empty")
            return

        while not self.isEmpty():
            print(self.arr[self.top], end=" ")
            self.top -= 1
        print()

               (a == '{' && b == '}') ||

                    ok = false;
                    break;

        }

        if (!stack.isEmpty())
            ok = false;

            System.out.println("Balanced");

        if (ch == '+' || ch == '-')
            return 1;

        if (ch == '*' || ch == '/')

                    ans += stack.pop();

                stack.push(ch);

            ans += stack.pop();

    }

public class Main {

    static int priority(char ch) {
        if (ch == '+' || ch == '-')
            return 1;
        if (ch == '*' || ch == '/')
            return 2;
        if (ch == '^')
            return 3;
        return 0;
    }

    static String reverse(String s) {

        char[] a = s.toCharArray();

        int i = 0;
        int j = a.length - 1;

        while (i < j) {
            char t = a[i];
            a[i] = a[j];
            a[j] = t;
            i++;
            j--;
        }

        return new String(a);
    }

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        String exp = sc.nextLine();

        exp = reverse(exp);

        char[] ch = exp.toCharArray();

        for (int i = 0; i < ch.length; i++) {

            if (ch[i] == '(')
                ch[i] = ')';

            else if (ch[i] == ')')
                ch[i] = '(';
        }

        Stack stack = new Stack(ch.length);

        String post = "";

        for (int i = 0; i < ch.length; i++) {

            char c = ch[i];

            if (Character.isLetterOrDigit(c))
                post += c;

            else if (c == '(')
                stack.push(c);

            else if (c == ')') {

                while (!stack.isEmpty() && stack.peek() != '(')
                    post += stack.pop();

                stack.pop();
            }

            else {

                while (!stack.isEmpty() &&
                        priority(stack.peek()) >= priority(c))
                    post += stack.pop();

                stack.push(c);
            }
        }

        while (!stack.isEmpty())
            post += stack.pop();

        System.out.println(reverse(post));
    }
}

        arr = new int[size];
        top = -1;
    }

    void push(int x) {
        arr[++top] = x;
    }

    int pop() {
        return arr[top--];
    }
}

public class Main {

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        String exp = sc.nextLine();

        Stack stack = new Stack(exp.length());

        for (int i = 0; i < exp.length(); i++) {

            char ch = exp.charAt(i);

            if (Character.isDigit(ch)) {
                stack.push(ch - '0');
            } else {

                int b = stack.pop();
                int a = stack.pop();
                int ans = 0;

                switch (ch) {

                    case '+':
                        ans = a + b;
                        break;

                    case '-':
                        ans = a - b;
                        break;

                    case '*':
                        ans = a * b;
                        break;

                    case '/':
                        ans = a / b;
                        break;

                    case '^':
                        ans = (int) Math.pow(a, b);
                        break;
                }

                stack.push(ans);
            }
        }

        System.out.println(stack.pop());
    }
}

import java.util.Scanner;

class TwoStacks {

    int arr[];
    int top1;
    int top2;
    int size;

    TwoStacks(int size) {
        this.size = size;
        arr = new int[size];
        top1 = -1;
        top2 = size;
    }

    void push1(int value) {

        if (top1 + 1 == top2) {
            System.out.println("Stack Overflow");
            return;
        }

        arr[++top1] = value;
    }

    void push2(int value) {

        if (top1 + 1 == top2) {
            System.out.println("Stack Overflow");
            return;
        }

        arr[--top2] = value;
    }

    void pop1() {

        if (top1 == -1) {
            System.out.println("Stack1 Empty");
            return;
        }

        System.out.println("Popped : " + arr[top1--]);
    }

    void pop2() {

        if (top2 == size) {
            System.out.println("Stack2 Empty");
            return;
        }

        System.out.println("Popped : " + arr[top2++]);
    }

    void peek1() {

        if (top1 == -1) {
            System.out.println("Stack1 Empty");
            return;
        }

        System.out.println(arr[top1]);
    }

    void peek2() {

        if (top2 == size) {
            System.out.println("Stack2 Empty");
            return;
        }

        System.out.println(arr[top2]);
    }

    void display1() {

        if (top1 == -1) {
            System.out.println("Stack1 Empty");
            return;
        }

        for (int i = top1; i >= 0; i--)
            System.out.print(arr[i] + " ");

        System.out.println();
    }

    void display2() {

        if (top2 == size) {
            System.out.println("Stack2 Empty");
            return;
        }

        for (int i = top2; i < size; i++)
            System.out.print(arr[i] + " ");

        System.out.println();
    }
}
