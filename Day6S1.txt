// Q1 : Population Migration
// Village starts with N people, every year 20% migrate, 15 arrive, every 4th year 10 extra leave
// Input : N (population), years -> Output : final population

// Python

n = int(input())
years = int(input())
i = 1
while i <= years:
    n = n - n * 20 // 100
    n += 15
    if i % 4 == 0:
        n -= 10
    i += 1
print(n)


// Java

import java.util.Scanner;
public class Q1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int years = sc.nextInt();
        for (int i = 1; i <= years; i++) {
            n = n - n * 20 / 100;
            n += 15;
            if (i % 4 == 0) n -= 10;
        }
        System.out.println(n);
    }
}

n = int(input())

while n > 1:

    cycles += 1
    if cycles % 3 == 0:
        n -= 1

    public static void main(String[] args) {

            if (i % 3 == 0) c -= 20;
            a = b;

        }
        System.out.println(b);
    }
}




// Q5 : Monster Population
// Start: 1, 3. new = prev + prev_prev, every prime generation 10% die
// Input : n -> Output : monsters at generation n

// Python

n = int(input())
vals = [0, 1, 3]

def isPrime(x):
    if x < 2:
        return False
    j = 2
    while j * j <= x:
        if x % j == 0:
            return False
        j += 1
    return True

i = 3
while i <= n:
    v = vals[i-1] + vals[i-2]
    if isPrime(i):
        v = v - v * 10 // 100
    vals.append(v)
    i += 1
print(vals[n])


// Java

import java.util.Scanner;
public class Q5 {
    static boolean isPrime(int x) {
        if (x < 2) return false;
        for (int j = 2; j * j <= x; j++)
            if (x % j == 0) return false;
        return true;
    }
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[] vals = new int[n + 1];
        vals[1] = 1;
        if (n >= 2) vals[2] = 3;
        for (int i = 3; i <= n; i++) {
            int v = vals[i-1] + vals[i-2];
            if (isPrime(i)) v = v - v * 10 / 100;
            vals[i] = v;
        }
        System.out.println(vals[n]);
    }
}




// Q6 : Galactic Fuel
// First 3 gens initialized, each new gen based on previous gens with hidden rules
// Input:1->1, Input:4->5, Input:8->56
// Pattern same as zombie: vals[i] = 2*vals[i-1] - vals[i-4], if %9==0 subtract 1

// Python

n = int(input())
vals = [1, 1, 2, 3]
if n <= 3:
    print(vals[n])
else:
    i = 4
    while i <= n:
        v = 2 * vals[i-1] - vals[i-4]
        if v % 9 == 0:
            v -= 1
        vals.append(v)
        i += 1
    print(vals[n])


// Java

import java.util.Scanner;
public class Q6 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[] vals = new int[n + 1];
        vals[0] = 1; vals[1] = 1; vals[2] = 2; vals[3] = 3;
        for (int i = 4; i <= n; i++) {
            int v = 2 * vals[i-1] - vals[i-4];
            if (v % 9 == 0) v -= 1;
            vals[i] = v;
        }
        System.out.println(vals[n]);
    }
}
